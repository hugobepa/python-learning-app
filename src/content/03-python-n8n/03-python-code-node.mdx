## 💻 Ejemplo Guiado

### Estructura Básica del Nodo Code

```python
# n8n proporciona 'items' como entrada
# Cada item tiene estructura: {"json": {...}}

# Ejemplo básico
output = []
for item in items:
    data = item['json']
    output.append({
        "json": {
            "procesado": True,
            "original": data,
            "timestamp": "2026-03-21"
        }
    })
return output
```

### Transformar Datos con Python

```python
# Procesar lista de usuarios
users = items[0]['json'].get('users', [])

processed = []
for user in users:
    processed.append({
        "json": {
            "id": user.get('id'),
            "nombre": user.get('nombre', '').upper(),
            "email": user.get('email', '').lower(),
            "activo": user.get('activo', False)
        }
    })

return processed
```

### Usar Librerías Externas

```python
import json
import re
from datetime import datetime

# Validar email con regex
def validar_email(email):
    pattern = r'^[\w\.-]+@[\w\.-]+\.\w+$'
    return bool(re.match(pattern, email))

# Procesar items
output = []
for item in items:
    data = item['json']
    email_valido = validar_email(data.get('email', ''))
    
    output.append({
        "json": {
            **data,
            "email_valido": email_valido,
            "procesado_en": datetime.now().isoformat()
        }
    })

return output
```

### Manejo de Errores

```python
try:
    # Procesar datos
    data = items[0]['json']
    resultado = data['valor'] / data['divisor']
    
    return [{
        "json": {
            "resultado": resultado,
            "estado": "success"
        }
    }]
except ZeroDivisionError:
    return [{
        "json": {
            "error": "División por cero",
            "estado": "failed"
        }
    }]
except KeyError as e:
    return [{
        "json": {
            "error": f"Campo faltante: {e}",
            "estado": "failed"
        }
    }]
```

## 🎮 Ejercicio Práctico

**Reto:** Procesador de datos de API

Crea un workflow que:
````markdown
1. Reciba datos de un webhook
2. Use nodo Code para validar y transformar
3. Filtre registros inválidos
4. Retorne datos procesados en formato estándar
````

-- **Input Ejemplo:**
```json
{
  "users": [
    {"id": 1, "nombre": "Alice", "email": "
```python
# Tu código aquí (nodo Code de n8n)
import json
from datetime import datetime

def procesar_usuario(user):
    """Valida y transforma un usuario."""
    # Validaciones
    if not user.get('nombre'):
        return None
    if not user.get('email') or '@' not in user['email']:
        return None
    if user.get('edad', 0) < 18:
        return None
    
    # Transformación
    return {
        "json": {
            "id": user.get('id'),
            "nombre": user['nombre'].strip().title(),
            "email": user['email'].strip().lower(),
            "edad": int(user['edad']),
            "procesado": True,
            "fecha_procesamiento": datetime.now().isoformat()
        }
    }

# Procesar todos los items
output = []
for item in items:
    user = item['json']
    procesado = procesar_usuario(user)
    if procesado:
        output.append(procesado)

return output
```
